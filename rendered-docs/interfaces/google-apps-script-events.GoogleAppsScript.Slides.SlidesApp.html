    Interface SlidesApp

    Creates and opens Presentations that can be edited.

    // Open a presentation by ID.
var preso = SlidesApp.openById('PRESENTATION_ID_GOES_HERE');

// Create and open a presentation.
preso = SlidesApp.create('Presentation Name');
