    Interface FormApp

    Allows a script to open an existing Form or create a new one.

    // Open a form by ID.
var existingForm = FormApp.openById('1234567890abcdefghijklmnopqrstuvwxyz');

// Create and open a form.
var newForm = FormApp.create('Form Name');
